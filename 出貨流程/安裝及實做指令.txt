常用指令：
sudo tpm2_nvdefine -C o -s 262 -a "ownerread|ownerwrite|policywrite"  0x01000001 
sudo tpm2_nvundefine 0x01000001
sudo tpm2_nvwrite -C o -i sig.rssa 0x01000001
sudo tpm2_changeauth -c o ownerauth
sudo tpm2_createprimary -C e -c primary.ctx 
sudo tpm2_create -G rsa -u rsa.pub -r rsa.priv -C primary.ctx
sudo tpm2_load -C primary.ctx -u rsa.pub -r rsa.priv -c rsa.ctx
sudo tpm2_evictcontrol -C o -c rsa.ctx 0x81010002
sudo tpm2_sign -c rsa.ctx -g sha256 -o sig.rssa CO.dat
sudo tpm2_verifysignature -c rsa.ctx -g sha256 -m CO.dat -s sig.rssa
